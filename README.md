# Implementing numeric types

Ints

a = 24  
b = 16  

print(a + b) # 40  
print(a > b) # True  
print(a < b) # False  

Floats  
FloatNum = 1.356  
IntNum = 4  
print(FloatNum + IntNum)  

# Theres no limit to the size of decimal, but python will round things up

one_third = 1 / 3  
print(one_third) # 0.3333333333333333  
print(one_third * 3) # 1.0
